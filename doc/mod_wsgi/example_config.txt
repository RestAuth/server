### This is an example apache configuration using mod_wsgi.

<VirtualHost *:443>
	# basic configuration...
	#...

        # basic django configuration:
        WSGIScriptAlias / /usr/share/restauth/wsgi/restauth

        <Location />
                AuthType Basic
                AuthName "RestAuth authentication service"
                AuthBasicProvider wsgi
                Require valid-user

                WSGIAuthUserScript /usr/share/restauth/wsgi/restauth
        </Location>
</VirtualHost>
